## 0、定义

- 时间复杂度：O(n^3)
- 求：从任意起点出发，到达任意起点的最短距离

```go
// Floyd 算法
func Floyd(arr [][]int) [][]int {
	n := len(arr)
	for k := 0; k < n; k++ {
		for i := 0; i < n; i++ {
			for j := 0; j < n; j++ {
				if arr[i][k]+arr[k][j] < arr[i][j] {
					arr[i][j] = arr[i][k] + arr[k][j]
				}
			}
		}
	}
	return arr
}
```
